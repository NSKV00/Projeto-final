CREATE TABLE funcionario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    contato VARCHAR(11) NOT NULL UNIQUE,
    cpf VARCHAR(14) NOT NULL UNIQUE,
    ativo VARCHAR(1) NOT NULL DEFAULT 'a'
);

CREATE TABLE cliente (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    telefone VARCHAR(11) NOT NULL UNIQUE,
    cpf VARCHAR(14) NOT NULL UNIQUE,
    cep VARCHAR(8) NOT NULL,
    ativo VARCHAR(1) NOT NULL DEFAULT 'a'
);

CREATE TABLE veiculos (
	id INT AUTO_INCREMENT PRIMARY KEY,
	cpf_cliente VARCHAR(14) NOT NULL,
	placa VARCHAR(7) NOT NULL UNIQUE,
	marca VARCHAR(100) NOT NULL,
	modelo VARCHAR(100) NOT NULL,
	ano_fabricacao VARCHAR(4) NOT NULL,
	FOREIGN KEY (cpf_cliente) REFERENCES cliente(cpf),
	ativo VARCHAR(1) NOT NULL DEFAULT 'a'
);

CREATE TABLE admin (
	id INT AUTO_INCREMENT PRIMARY KEY,
	usuario VARCHAR(100) NOT NULL,
	senha VARCHAR(8) NOT NULL
);

CREATE TABLE os (
	id INT AUTO_INCREMENT PRIMARY KEY,
	cpf_cliente VARCHAR(14) not null,
	FOREIGN KEY (cpf_cliente) REFERENCES cliente(cpf),
	placa_veiculo VARCHAR(7) not null,
	FOREIGN KEY (placa_veiculo) REFERENCES veiculos(placa),
	cpf_funcionario VARCHAR(14) not null,
	FOREIGN KEY (cpf_funcionario) REFERENCES funcionario(cpf),
	problema VARCHAR(599) NOT NULL,
	data_os datetime not null,
	servico VARCHAR(599) NOT NULL,
	ativo VARCHAR(1) NOT NULL DEFAULT 'a'
);
